CHECK EYES (Look for Face landmarks) if (results.multiFaceLandmarks && results.multiFaceLandmarks[0]) { const lm = results.multiFaceLandmarks[0]; // EAR calculation (Eye Aspect Ratio) const p = [33, 160, 158, 133, 153, 144]; const ear = (Math.hypot(lm[160].x-lm[144].x, lm[160].y-lm[144].y) + Math.hypot(lm[158].x-lm[153].x, lm[158].y-lm[153].y)) / (2 * Math.hypot(lm[33].x-lm[133].x, lm[33].y-lm[133].y)); eyesClosed = ear < 0.20; // Lowered threshold slightly for better detection document.getElementById('eyes').textContent = eyesClosed ? "✅ Closed" : "⚠️ OPEN"; document.getElementById('eyes').className = eyesClosed ? "good" : "warning-text"; } else { // If no face is detected, treat as a violation eyesClosed = false; document.getElementById('eyes').textContent = "❌ FACE NOT FOUND"; } // 2. "✅ Upright" : "⚠️ SLUMPING"; document.getElementById('spine').className = postureGood ? "good" : "warning-text"; } // 3. ENFORCE RULES if (!eyesClosed || !postureGood) { // If either rule is broken, start counting towards violation if (!eyesClosed) eyesOpenTime += delta * 1000; if (!postureGood) badPostureTime += delta * 1000; // Show warnings in the status bar statusEl.textContent = "🛑 VIOLATION DETECTED! FIX POSE NOW!"; statusEl.className = "violation"; } else { // Reset timers if they are behaving eyesOpenTime = 0; badPostureTime = 0; statusEl.textContent = "🟢 Mining Active — Stay Still"; statusEl.className = ""; } // 4. TRIGGER AUTO-END if (eyesOpenTime > 4000 || badPostureTime > 4000) { console.log("Auto-ending due to cheating/violation"); processSessionEnd("Security Kick: Rules Not Followed"); } }
